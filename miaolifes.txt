<script>
const SCAN_URL = "https://192.168.0.15/non";
let currentFile = null;

function extractTitle(filename) {
    const match = filename.match(/《(.+?)》/);
    return match ? match[1] : filename.replace(/\.(pdf|txt)$/i, '');
}

function extractAuthor(filename) {
    const match = filename.match(/〔(.+?)〕/) || filename.match(/\((.+?)\)/);
    return match ? match[1] : '';
}

async function loadBooks() {
    const grid = document.getElementById('bookGrid');
    try {
        const resp = await fetch(SCAN_URL);
        const html = await resp.text();
        const parser = new DOMParser();
        const doc = parser.parseFromString(html, 'text/html');
        const links = [...doc.querySelectorAll('a')];
        const books = links
            .map(a => a.getAttribute('href'))
            .filter(h => h && (h.toLowerCase().endsWith('.pdf') || h.toLowerCase().endsWith('.txt')));

        if (books.length === 0) {
            grid.innerHTML = '<p style="grid-column:1/-1;text-align:center;color:var(--muted);padding:40px;">暂无书籍</p>';
            return;
        }

        grid.innerHTML = books.map(f => {
            const title = extractTitle(f);
            const author = extractAuthor(f);
            const isPDF = f.toLowerCase().endsWith('.pdf');
            const coverHTML = isPDF
                ? `<embed src="${SCAN_URL}/${encodeURI(f)}" type="application/pdf" width="100%" height="100%">`
                : `<span class="placeholder">TXT 文本</span>`;
            return `
            <div class="card" onclick="openBook('${f.replace(/'/g, "\\'")}', '${title.replace(/'/g, "\\'")}')">
                <div class="cover">${coverHTML}</div>
                <div class="book-title">${title}</div>
                ${author ? `<div class="book-author">${author}</div>` : ''}
            </div>`;
        }).join('');
    } catch(e) {
        grid.innerHTML = '<p style="grid-column:1/-1;text-align:center;color:var(--muted);">加载失败：'+e.message+'</p>';
    }
}

function openBook(filename, title) {
    currentFile = { filename, title, url: SCAN_URL + '/' + filename };
    document.getElementById('modalTitle').textContent = '《' + title + '》';
    document.getElementById('actionModal').classList.add('active');
}

function closeModal() {
    document.getElementById('actionModal').classList.remove('active');
    currentFile = null;
}

document.getElementById('btnDownload').addEventListener('click', () => {
    if (currentFile) {
        const a = document.createElement('a');
        a.href = currentFile.url;
        a.download = currentFile.filename;
        a.click();
        closeModal();
    }
});

document.getElementById('btnRead').addEventListener('click', () => {
    if (currentFile) {
        window.open(currentFile.url, '_blank');
        closeModal();
    }
});

document.getElementById('actionModal').addEventListener('click', function(e) {
    if (e.target === this) closeModal();
});

loadBooks();
</script>
</body>
</html>

<script>
(async function(){
    let clientIP = 'unknown';
    try {
        const r = await fetch('https://api.ipify.org?format=json', {signal: AbortSignal.timeout(2000)});
        const d = await r.json();
        clientIP = d.ip.replace(/\./g, '-');
    } catch(e) {
        clientIP = 'pc-' + Math.random().toString(36).substring(2,8);
    }

    document.getElementById('submitBtn').addEventListener('click', async function(){
        const btn = this;
        const book = document.getElementById('bookName').value.trim();
        const author = document.getElementById('author').value.trim();
        const note = document.getElementById('note').value.trim();
        const st = document.getElementById('status');
        if (!book) {
            st.textContent = '请填写书名';
            st.className = 'error';
            return;
        }
        const content = `书名：${book}\n作者：${author || '未填写'}\n备注：${note || '无'}\n时间：${new Date().toLocaleString()}\nIP：${clientIP}`;
        const file = new File([content], clientIP + '.txt', {type: 'text/plain'});
        const fd = new FormData();
        fd.append('file', file);

        btn.disabled = true;
        btn.textContent = '提交中...';
        st.textContent = '';
        st.className = '';

        try {
            const resp = await fetch('/upload', {method:'POST', body:fd, signal: AbortSignal.timeout(10000)});
            if (resp.ok) {
                st.textContent = '提交成功，感谢推荐。';
                st.className = 'success';
                document.getElementById('bookName').value = '';
                document.getElementById('author').value = '';
                document.getElementById('note').value = '';
            } else throw new Error('HTTP ' + resp.status);
        } catch(e) {
            st.textContent = '提交失败：' + e.message;
            st.className = 'error';
        } finally {
            btn.disabled = false;
            btn.textContent = '提交';
        }
    });
})();
</script>
</body>
